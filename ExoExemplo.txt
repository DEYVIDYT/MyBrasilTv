	@Override
	public void onClick(View _view) {
		try {
			videoURL = edittext1.getText().toString();
			subtitleURL = edittext2.getText().toString();
			playerView.setPlayer(exoPlayer);
			Uri assetSrtUri = Uri.parse(subtitleURL);
			MediaItem.SubtitleConfiguration subtitle = new MediaItem.SubtitleConfiguration.Builder(assetSrtUri)
			                .setMimeType(MimeTypes.APPLICATION_SUBRIP)
			                .setLanguage("fa")
			                .setSelectionFlags(C.SELECTION_FLAG_DEFAULT)
			                .build();
			List<MediaItem.SubtitleConfiguration> subtitleConfigurations = new ArrayList<>();
			        subtitleConfigurations.add(subtitle);
			Uri assetVideoUri = Uri.parse(videoURL);
			MediaItem mediaItem = new MediaItem.Builder()
			                .setUri(assetVideoUri)
			                .setSubtitleConfigurations(subtitleConfigurations)
			                .build();
			exoPlayer.setTrackSelectionParameters(trackSelectionParameters);
			exoPlayer.setMediaItem(mediaItem);
			exoPlayer.prepare();
			exoPlayer.play();
			
			button1.setEnabled(true);
		} catch (Exception e) {
			
		}
	}
});
